rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    match /conversations/{cid} {
      // Read: Allow if signed in (query already filters by participants)
      // For collection queries with where("participants", "array-contains", uid), 
      // Firestore only returns conversations where user is participant
      allow read: if isSignedIn();

      // Create: Allow if user is a participant (UI ensures they're friends)
      allow create: if
        isSignedIn()
        &&
        request.resource.data.participants is list
        &&
        request.resource.data.participants.size() == 2
        &&
        request.auth.uid in request.resource.data.participants;

      // Update: Allow if user is a participant
      allow update: if
        isSignedIn()
        &&
        resource.data.participants is list
        &&
        request.auth.uid in resource.data.participants;

      match /messages/{mid} {
        function parentConversation() {
          return get(/databases/$(database)/documents/conversations/$(cid));
        }

        // Read messages if signed in (parent conversation access already verified)
        allow read: if isSignedIn();

        // Create message if user is sender
        allow create: if isSignedIn()
          && request.resource.data.senderId == request.auth.uid;

        // Update: Allow updating readBy field if user is a participant in parent conversation
        // Note: Frontend only updates readBy field, so we trust that here
        allow update: if isSignedIn()
          && request.auth.uid in parentConversation().data.participants;
      }

      match /typing/{userId} {
        function parentConversation() {
          return get(/databases/$(database)/documents/conversations/$(cid));
        }

        // Read typing status: allow if participant
        allow read: if isSignedIn() 
          && request.auth.uid in parentConversation().data.participants;

        // Write typing status: allow if updating own status and is participant
        allow write: if isSignedIn() 
          && request.auth.uid == userId
          && request.auth.uid in parentConversation().data.participants;
      }
    }

    match /friendships/{userId}/friends/{friendId} {
      // Allow users to read their own friendships
      allow read: if isSignedIn() 
        && (request.auth.uid == userId || request.auth.uid == friendId);
      // Only backend can write friendships (via Admin SDK)
      allow write: if false;
    }
  }
}
